POST /api/v1/players
Create Player

Add a new player

Examples

REQUEST
{
  "player": {
    "id": 9495,
    "name_brief": "N. P.",
    "first_name": "New",
    "last_name": "Player",
    "position": "SP",
    "age": 20,
    "average_position_age_diff": "-6"
  }
} 

RESPONSE
{
  "message": "Success",
  "success": true,
  "status": 200,
  "player": {
    "id": 9495,
    "name_brief": "N. P.",
    "first_name": "New",
    "last_name": "Player",
    "position": "SP",
    "age": 20,
    "average_position_age_diff": "-6"
  }
}
